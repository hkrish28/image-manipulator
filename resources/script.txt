save resources/res/bird-green-split.jpg bird-green
rgb-combine bird-combined bird-red bird-green bird-blue
save resources/res/bird-combined.jpg bird-combined

blur bird bird-blur
save resources/res/bird-blur.jpg bird-blur

sharpen bird bird-sharpen
save resources/res/bird-sharpen.jpg bird-sharpen

sepia bird bird-sepia
save resources/res/bird-sepia.jpg bird-sepia

#load bird.png and call it 'bird2'
load resources/images/bird.png bird2

#brighten bird by adding 10
brighten 10 bird2 bird-brighter

#flip bird vertically
vertical-flip bird2 bird-vertical

#flip the vertically flipped bird horizontally
horizontal-flip bird-vertical bird-vertical-horizontal

save resources/res/bird-vertical-horizontal.jpg bird-vertical-horizontal
save resources/res/bird-vertical.jpg bird-vertical

#create a greyscale using only the value component, as an image bird-greyscale
value-component bird2 bird-greyscale

luma-component bird2 bird-luma

intensity-component bird2 bird-intensity

#save bird-brighter
save resources/res/bird-brighter.jpg bird-brighter

#save bird-greyscale
save resources/res/bird-value-gs.jpg bird-greyscale
save resources/res/bird-luma.jpg bird-greyscale
save resources/res/bird-intensity.jpg bird-intensity

#give the bird a red tint

#brighten just the red image
brighten 50 bird-red bird-red

#combine them back, but by using the brightened red we get a red tint
rgb-combine bird-red-tint bird-red bird-green bird-blue
save resources/res/bird-red-tint.jpg bird-red-tint

#overwrite bird with another file
load resources/images/bird.png bird

brighten -25 bird bird-dark

save resources/res/bird-darker.jpg bird-dark
